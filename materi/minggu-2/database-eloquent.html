2.5 Database & Eloquent

📘 Minggu 2 ⏱️ 30 menit

🎯 Tujuan Belajar

  • Membuat migration untuk tabel database
  • Menggunakan Eloquent ORM untuk CRUD
  • Menghubungkan Livewire dengan database

1. Buat Migration

Terminal 
php artisan make:model Todo -m
create_todos_table.php 
Schema::create('todos', function (Blueprint $table) {
    $table->id();
    $table->string('task');
    $table->boolean('is_done')->default(false);
    $table->timestamps();
});
Terminal
php artisan migrate

2. Setup Model

app/Models/Todo.php 
class Todo extends Model
{
    protected $fillable = ['task', 'is_done'];
}

3. CRUD dengan Eloquent

TodoList.php (Livewire) 
use App\Models\Todo;

class TodoList extends Component
{
    public $task = '';
    
    // READ - Ambil semua data
    public function render()
    {
        return view('livewire.todo-list', [
            'todos' => Todo::latest()->get()
        ]);
    }
    
    // CREATE - Simpan data baru
    public function addTodo()
    {
        Todo::create(['task' => $this->task]);
        $this->task = '';
    }
    
    // UPDATE - Ubah status
    public function toggleTodo($id)
    {
        $todo = Todo::find($id);
        $todo->update(['is_done' => !$todo->is_done]);
    }
    
    // DELETE - Hapus data
    public function deleteTodo($id)
    {
        Todo::find($id)->delete();
    }
}
💡

Eloquent Cheat Sheet

Model::all() = Ambil semua
Model::find($id) = Cari by ID
Model::create([]) = Buat baru
$model->update([]) = Update
$model->delete() = Hapus
