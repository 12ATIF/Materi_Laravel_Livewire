3.2 Pagination & Search

📕 Minggu 3 ⏱️ 25 menit

🎯 Tujuan Belajar

  • Menampilkan data dengan pagination
  • Membuat live search tanpa refresh
  • Kombinasi search + pagination

1. Pagination

BookList.php 
use Livewire\WithPagination;

class BookList extends Component
{
    use WithPagination;
    
    public function render()
    {
        return view('livewire.book-list', [
            'books' => Book::paginate(10)
        ]);
    }
}
book-list.blade.php 
@foreach($books as $book)
    <div>{{ $book->title }}</div>
@endforeach

<!-- Pagination links -->
{{ $books->links() }}

2. Live Search

BookList.php 
use Livewire\WithPagination;

class BookList extends Component
{
    use WithPagination;
    
    public $search = '';
    
    // Reset ke halaman 1 saat search berubah
    public function updatingSearch()
    {
        $this->resetPage();
    }
    
    public function render()
    {
        return view('livewire.book-list', [
            'books' => Book::where('title', 'like', '%'.$this->search.'%')
                          ->paginate(10)
        ]);
    }
}
book-list.blade.php 
<!-- Search input -->
<input type="text" 
       wire:model.live.debounce.300ms="search"
       placeholder="Cari buku...">

<!-- Results -->
@forelse($books as $book)
    <div>{{ $book->title }}</div>
@empty
    <p>Tidak ada buku ditemukan.</p>
@endforelse

{{ $books->links() }}
💡

Debounce

debounce.300ms = tunggu 300ms setelah user berhenti mengetik baru kirim request. Hemat server!
