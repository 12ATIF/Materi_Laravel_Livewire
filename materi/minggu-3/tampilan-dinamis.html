3.4 Tampilan Dinamis

📕 Minggu 3 ⏱️ 25 menit

🎯 Tujuan Belajar

  • Membuat modal dengan Livewire
  • Menampilkan loading state
  • Flash message / toast notification

1. Modal

BookManager.php 
public $showModal = false;
public $selectedBook;

public function openModal($bookId)
{
    $this->selectedBook = Book::find($bookId);
    $this->showModal = true;
}

public function closeModal()
{
    $this->showModal = false;
    $this->selectedBook = null;
}
book-manager.blade.php 
<!-- Tombol buka modal -->
<button wire:click="openModal({{ $book->id }})">Detail</button>

<!-- Modal -->
@if($showModal)
<div class="modal-overlay">
    <div class="modal">
        <h2>{{ $selectedBook->title }}</h2>
        <p>{{ $selectedBook->description }}</p>
        <button wire:click="closeModal">Tutup</button>
    </div>
</div>
@endif

2. Loading State

Loading States 
<!-- Loading saat submit -->
<button wire:click="save">
    <span wire:loading.remove wire:target="save">Simpan</span>
    <span wire:loading wire:target="save">Menyimpan...</span>
</button>

<!-- Disable button saat loading -->
<button wire:loading.attr="disabled">Simpan</button>
